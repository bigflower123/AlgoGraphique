484 void Piloter::OnIdle(wxIdleEvent & event)
485 {
486  static int betweenRefreshSlice = 0;
487  static int betweenRefreshSliceCount = 0;
488  const long timeSlice = 1000/25; // 25 img/seconde max
489 
490  wxDateTime time = wxDateTime::UNow();
491  wxTimeSpan delta = time.Subtract(lastTime);
492  long milli = delta.GetMilliseconds().ToLong();
493 
494  if (milli >= timeSlice)
495  {
496  betweenRefreshSlice += (int)milli;
497  betweenRefreshSliceCount++;
498 
499  if (betweenRefreshSliceCount % 10== 0)
500  {
501  wxString msg;
502  msg.Printf(wxT("%.2lf msec/frame %.2lf frames/sec"),
503  betweenRefreshSlice / (double)betweenRefreshSliceCount,
504  1000 * betweenRefreshSliceCount
505  / (double) betweenRefreshSlice );
506  status->SetLabel(msg);
507  betweenRefreshSlice = 0;
508  betweenRefreshSliceCount = 0;
509  }
510  lastTime = time;
511  engine->Refresh();
512  }
513  else
514  {
515  // on est trop rapide => on attend un peu
516  wxMilliSleep(timeSlice - milli);
517  }
518  event.RequestMore(); // on demande d'autres evenements idle
519 }
